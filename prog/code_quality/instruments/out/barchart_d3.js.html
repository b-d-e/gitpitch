instruments = [{name:"concertina", number: 2},
               {name:"piano", number: 5},

/** make the viz 
 - @param scale x multiplier
*/
function doit(scale){
    d3.select("body")
        .selectAll("rect")
        .data(instruments)
        .attr("width", function(d) { console.log("name " + d.name);return d.number*scale; })
}
